Useful command

yum package manager
yum provides
yum history
yum install {package_name}
yum remove {package_name}
yum update
yum whatprovides {package_name}
yum list installed
yum-config-manager --enable {package_name}
yum check-update
yum -y install {http .rpm link}
yum makecache fast
# To exclude list of packages, use the command as follows.
yum --exclude=mysql\* --exclude=httpd\* update
Static route
#Add route for a network
route add -net 10.0.0.0/8 gw 192.168.219.1
(This will only add route temporarly, add the command in '/etc/sysconfig/network-scripts/route-eth0' for permanent)

#Add route for a host
route add host 10.10.15.17 gw 192.168.219.2

#Delete route host
route del host 10.10.15.17

# see routing table
ip route
route -n

#Add route in windows server (persistent)
route -p ADD 192.168.35.0 MASK 255.255.255.0 192.168.0.2
Networking in debian
#Change ip address in debian
vi /etc/network/interfaces

#Restart network
/etc/init.d/networking restart
File transfer (sftp)
#File transfer in different port
sftp -P 8079 username@ipaddress
yes | no to the question
#say yes to copy overwrite
yes | cp source_file destination_file

cat <<EOF>>/path/to/file

User management
#Delete user along with home directory
userdel -r {username}
RPM
#Query Package
rpm -qa package_name

#Install Package Name
rpm -ivh package_name

#Delete Package name
rpm -e package_name

#Get time of kernel update
rpm -qa --last |grep kernel
Swap and buffer space
#Clear swap memory
swapoff -a
swapon -a

#clear buffer cache
sync; echo 1 > /proc/sys/vm/drop_caches
sync; echo 2 > /proc/sys/vm/drop_caches
sync; echo 3 > /proc/sys/vm/drop_caches
Networking
#Disable IPV6 add in /etc/sysctl.conf
net.ipv6.conf.all.disable_ipv6 = 1
net.ipv6.conf.default.disable_ipv6 = 1

sysctl -p
CPU core
calculate total logical core

The total number of logical cores = CPU sockets × physical cores per socket × threads per physical core.
Therefore, the computer has 4 * 1 * 1 = 4 logical cores in total.
Process Management
ps aux
	a = all users
	u = list detail information
	x = include boot process
load Average calculation
SSH
#To remove 'Remote host identification has changed'
ssh-keygen -R host_ip_address
Rsync
rsync -rtv source destination_folder

r = recursive into directories
t = preserve modification time
v = verbose
Teaming
[root@node1 ~]# cat /etc/sysconfig/network-scripts/ifcfg-em1
TYPE=Ethernet
NAME=em1
DEVICE=em1
ONBOOT=yes
MASTER=bond0
SLAVE=yes

[root@node1~]# cat /etc/sysconfig/network-scripts/ifcfg-em2
TYPE=Ethernet
NAME=em2
DEVICE=em2
ONBOOT=yes
MASTER=bond0
SLAVE=yes

[root@node1]# cat /etc/sysconfig/network-scripts/ifcfg-bond0

BONDING_OPTS="miimon=1 updelay=0 downdelay=0 mode=active-backup" TYPE=Bond

BOOTPROTO=static
IPADDR=10.20.20.50
PREFIX=24
GATEWAY=10.20.20.1
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
